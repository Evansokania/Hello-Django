from django.shortcuts import render, redirect, get_object_or_404 from django.contrib.auth.decorators import login_required from django.http import HttpResponseForbidden from .models import Employee @login_required def delete_employee(request, pk): if not request.user.is_admin: # Only admins can delete return HttpResponseForbidden("You are not allowed to delete employees.") employee = get_object_or_404(Employee, pk=pk) if request.method == "POST": employee.delete() return redirect("employee-list") return render(request, "myapp/delete_employee.html", {"employee": employee})
